class SleepDataset(Dataset):
    def __init__(self, x_paths, y_paths):
        self.x_data = []
        self.y_data = []
        self.samples = []
        self.all_labels = []

        print("Preloading all data into memory...")
        for f_idx, (x_p, y_p) in enumerate(tqdm(zip(x_paths, y_paths), total=len(x_paths))):
            x = np.load(x_p).astype(np.float32)
            y = np.load(y_p)
            self.x_data.append(x)
            self.y_data.append(y)
            for e_idx, lbl in enumerate(y):
                self.samples.append((f_idx, e_idx))
                self.all_labels.append(lbl)
        print(f"✅ Done. {len(self.samples)} samples loaded into memory.")

    def __len__(self):
        return len(self.samples)

    def __getitem__(self, idx):
        f_idx, e_idx = self.samples[idx]
        x = torch.from_numpy(self.x_data[f_idx][e_idx])
        y = torch.tensor(self.y_data[f_idx][e_idx], dtype=torch.long)
        return x, y

def show_proportions(dataset, title="Dataset Distribution"):
    labels = np.array(dataset.all_labels)
    total = len(labels)

    # Count occurrences (0=W, 1=N1, 2=N2, 3=N3, 4=R)
    counts = np.bincount(labels)
    stages = ['Wake', 'N1', 'N2', 'N3', 'REM']

    print(f"--- {title} ---")
    for i, count in enumerate(counts):
        percent = (count / total) * 100
        # Check to ensure we don't index out of bounds if stages are missing
        stage_name = stages[i] if i < len(stages) else f"Stage {i}"
        print(f"{stage_name:<5}: {count:<6} samples ({percent:>5.2f}%)")

    # Visualization
    plt.figure(figsize=(8, 5))
    colors = ['#ff9999','#66b3ff','#99ff99','#ffcc99', '#c2c2f0']
    
    # We use counts[:len(stages)] to match the labels provided
    plt.pie(counts, labels=stages[:len(counts)], autopct='%1.1f%%', startangle=140, colors=colors)
    plt.title(title)
    plt.show()

import random
def physical_downsample(dataset, wake_keep_ratio=0.15):
    """
    Physically removes a portion of Wake samples from the dataset index.
    """
    new_samples = []
    new_labels = []

    # Use a fixed seed so your results are reproducible
    random.seed(42)

    for i in range(len(dataset.samples)):
        f_idx, e_idx = dataset.samples[i]
        label = dataset.all_labels[i]

        if label == 0:  # Wake Stage
            if random.random() < wake_keep_ratio:
                new_samples.append((f_idx, e_idx))
                new_labels.append(label)
        else:
            # Keep 100% of all other stages (N1, N2, N3, REM)
            new_samples.append((f_idx, e_idx))
            new_labels.append(label)

    # Update the dataset with the filtered lists
    dataset.samples = new_samples
    dataset.all_labels = new_labels

    print(f"✅ Downsampling Complete!")
    print(f"New total training samples: {len(dataset.samples)}")

class_counts = torch.tensor([31295, 13174, 49574, 10702, 19026], dtype=torch.float32)
class_weights = 1.0 / class_counts

class SleepStageCNN_LSTM(L.LightningModule):
    def __init__(self, learning_rate=1e-3):
        super().__init__()
        self.save_hyperparameters()

        # 1. CNN Feature Extractor
        # Input shape: (Batch, 2 Channels, 3000 Samples)
        self.features = nn.Sequential(
            # First layer looks for broad patterns
            nn.Conv1d(2, 64, kernel_size=50, stride=5),
            nn.BatchNorm1d(64),
            nn.ReLU(),
            nn.MaxPool1d(kernel_size=8),

            # Second layer looks for more complex combinations
            nn.Conv1d(64, 128, kernel_size=8, stride=1),
            nn.BatchNorm1d(128),
            nn.ReLU(),
            nn.MaxPool1d(kernel_size=4),

            # Third layer compresses the features
            nn.Conv1d(128, 128, kernel_size=3, stride=1),
            nn.BatchNorm1d(128),
            nn.ReLU(),
        )

        # 2. LSTM for temporal context
        # We treat the 128 features as a single time-step vector
        self.lstm = nn.LSTM(input_size=128, hidden_size=128, 
                            num_layers=2, batch_first=True, dropout=0.3)
        # 3. Output Classifier
        self.dropout = nn.Dropout(0.5)
        self.classifier = nn.Linear(128, 5) # 5 Classes: W, N1, N2, N3, REM

        # Metrics
        self.val_f1 = MulticlassF1Score(num_classes=5, average='macro')
        self.val_acc = MulticlassAccuracy(num_classes=5)

    def forward(self, x):
        # CNN: (Batch, 2, 3000) -> (Batch, 128)
        x = self.features(x)

        # Transpose for LSTM: (Batch, T, 128)
        x = x.permute(0, 2, 1)

        # LSTM processes the sequence, take the last hidden state
        out, (hn, cn) = self.lstm(x)
        x = hn[-1]  # Last hidden state, XLA-friendly

        x = self.dropout(x)
        
        return self.classifier(x)

    def training_step(self, batch, batch_idx):
        x, y = batch
        logits = self(x)
        weights = class_weights.to(x.device)  # Move to TPU/GPU
        loss = F.cross_entropy(logits, y, weight=weights)
        self.log("train_loss", loss, prog_bar=True, on_step=False, on_epoch=True)
        return loss

    def validation_step(self, batch, batch_idx):
        x, y = batch
        logits = self(x)
        loss = F.cross_entropy(logits, y)

        # Log F1 and Accuracy
        self.log_dict({
            "val_loss": loss,
            "val_f1": self.val_f1(logits, y),
            "val_acc": self.val_acc(logits, y)
        }, prog_bar=True)

    def configure_optimizers(self):
        optimizer = torch.optim.Adam(self.parameters(), lr=self.hparams.learning_rate)
        scheduler = torch.optim.lr_scheduler.CosineAnnealingLR(
            optimizer, 
            T_max=9,       # Match your max_epochs
            eta_min=1e-5   # Minimum LR it decays to
        )
        return {
            "optimizer": optimizer,
            "lr_scheduler": {
                "scheduler": scheduler,
                "interval": "epoch"
            }
        }
